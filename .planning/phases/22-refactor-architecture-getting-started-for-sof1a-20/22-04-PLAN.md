<task type="auto">
  <name>Task 2: Verify navigation consistency and link integrity</name>
  <files>docs/docs.json, docs/getting-started/*.mdx, docs/architecture/*.mdx</files>
  <action>
    Perform a full link-integrity check across the modified getting-started and architecture pages:
    1. Extract all internal href="/..." references from the 12 modified MDX files
    2. For each href path, check if the target .mdx file exists under docs/
    3. Also check that docs.json navigation entries for "getting-started" and "architecture" groups point to files that exist
    4. Report any broken links in a summary
    Note: Some hrefs may point to anchor IDs (e.g., /architecture/core-components#citadel-gateway) — ignore the anchor for file-existence checks.
  </action>
  <verify>
    MISSING=$(grep -roP 'href="/[^"]+"' docs/getting-started/*.mdx docs/architecture/*.mdx | \
    sed 's|href="/||; s|"$||' | \
    awk -F'#' '{print $1}' | \
    sort -u | \
    awk '{print "docs/" $0 ".mdx"}' | \
    while read f; do [ -f "$f" ] || echo "MISSING: $f"; done)
    
    NAV_MISSING=$(node -e "
      const fs = require('fs');
      const docs = JSON.parse(fs.readFileSync('docs/docs.json', 'utf8'));
      function checkPages(pages, prefix) {
        for (const p of pages) {
          if (typeof p === 'string') {
            const path = 'docs/' + p + '.mdx';
            if (!fs.existsSync(path)) console.log('MISSING_NAV: ' + path);
          } else if (p.pages) {
            checkPages(p.pages, prefix);
          }
        }
      }
      checkPages(docs.navigation.groups.flatMap(g => g.pages));
    " 2>/dev/null)
    
    TOTAL_MISSING=$(echo "$MISSING $NAV_MISSING" | grep -c "MISSING" || true)
    [ "$TOTAL_MISSING" -eq 0 ]
  </verify>
  <done>All internal links and navigation entries resolve to existing files</done>
</task>

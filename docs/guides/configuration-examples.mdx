# Configuration Examples

This guide provides real-world configuration examples for every major component of the AI Landing Zone (Citadel Governance Hub). All examples are production-tested and include inline explanations of required vs. optional parameters.

<Note>
All examples use placeholder values (`<your-value>`). Replace with environment-specific values before deployment. See the [Deployment Guide](/guides/deployment-guide) for context on where each configuration applies.

## Network Configuration Examples

### Hub VNet with Multiple Spokes

The hub VNet hosts shared services (APIM, monitoring, private DNS). Spoke VNets host individual AI workloads.

```bicep
// hub-vnet.bicep — Required parameters marked with *
param location string = 'eastus'                   // * Azure region
param hubVnetName string = 'vnet-hub-ai-prod-eus'  // * Hub VNet name
param hubAddressPrefix string = '10.0.0.0/16'      // * Hub address space

resource hubVnet 'Microsoft.Network/virtualNetworks@2023-09-01' = {
  name: hubVnetName
  location: location
  properties: {
    addressSpace: {
      addressPrefixes: [hubAddressPrefix]
    }
    subnets: [
      {
        name: 'snet-apim'                    // * Required: APIM subnet
        properties: {
          addressPrefix: '10.0.1.0/24'
          serviceEndpoints: [
            { service: 'Microsoft.KeyVault' }
            { service: 'Microsoft.Storage' }
          ]
          networkSecurityGroup: { id: nsgApim.id }
        }
      }
      {
        name: 'snet-private-endpoints'       // * Required: Private endpoint subnet
        properties: {
          addressPrefix: '10.0.2.0/24'
          privateEndpointNetworkPolicies: 'Disabled'  // Required for private endpoints
        }
      }
      {
        name: 'snet-management'              // Optional: Bastion/management
        properties: {
          addressPrefix: '10.0.3.0/24'
        }
      }
      {
        name: 'AzureBastionSubnet'           // Optional: Named exactly for Bastion
        properties: {
          addressPrefix: '10.0.4.0/26'       // /26 minimum for Bastion
        }
      }
    ]
  }
}

// Spoke VNet — Deploy once per AI workload team
resource spokeVnet 'Microsoft.Network/virtualNetworks@2023-09-01' = {
  name: 'vnet-spoke-hr-ai-prod-eus'
  location: location
  properties: {
    addressSpace: {
      addressPrefixes: ['10.1.0.0/24']      // Non-overlapping with hub and other spokes
    }
    subnets: [
      {
        name: 'snet-ai-workload'
        properties: {
          addressPrefix: '10.1.0.0/25'
          delegations: [
            {
              name: 'delegation-aci'
              properties: {
                serviceName: 'Microsoft.ContainerInstance/containerGroups'
              }
            }
          ]
        }
      }
    ]
  }
}

// VNet Peering — Hub-to-spoke (repeat for each spoke)
resource hubToSpokePeering 'Microsoft.Network/virtualNetworks/virtualNetworkPeerings@2023-09-01' = {
  name: '${hubVnet.name}/peer-to-spoke-hr'
  properties: {
    remoteVirtualNetwork: { id: spokeVnet.id }
    allowVirtualNetworkAccess: true
    allowForwardedTraffic: true
    allowGatewayTransit: true              // Hub provides gateway transit to spokes
    useRemoteGateways: false
  }
}
```

```bicep
// private-dns-zones.bicep — Deploy in hub subscription
var aiDnsZones = [
```

### Rate Limiting Policy Template

```xml
<!-- rate-limiting-policy.xml — Apply at product or API level -->
<policies>
  <inbound>
    <base />
    <!-- Token-based rate limiting for AI workloads -->
    <azure-openai-token-limit
      counter-key="@(context.Subscription.Id)"
      tokens-per-minute="100000"           <!-- Adjust per workload SLA -->
      estimate-prompt-tokens="true"
      tokens-consumed-header-name="x-tokens-consumed"
      remaining-tokens-header-name="x-remaining-tokens" />

    <!-- Request rate limiting (fallback) -->
    <rate-limit-by-key
      calls="60"
      renewal-period="60"
      counter-key="@(context.Subscription.Id)"
      increment-condition="@(context.Response.StatusCode == 200)"
      remaining-calls-header-name="x-ratelimit-remaining"
      retry-after-header-name="Retry-After" />
  </inbound>
  <backend>
    <base />
  </backend>
  <outbound>
    <base />
  </outbound>
  <on-error>
    <base />
    <return-response>
      <set-status code="429" reason="Too Many Requests" />
      <set-header name="Retry-After" exists-action="override">
        <value>@(context.Response.Headers.GetValueOrDefault("Retry-After", "60"))</value>
      </set-header>
      <set-body>{"error": {"code": "429", "message": "Rate limit exceeded. Retry after indicated seconds."}}</set-body>
    </return-response>
  </on-error>
</policies>
```

### Entra ID Authentication Policy

```xml
<!-- entra-auth-policy.xml — JWT validation for Entra ID tokens -->
<policies>
  <inbound>
    <base />
    <validate-jwt
      header-name="Authorization"
      failed-validation-httpcode="401"
      failed-validation-error-message="Unauthorized: Invalid or missing token"
      require-expiration-time="true"
      require-scheme="Bearer"
      require-signed-tokens="true">
      <openid-config url="https://login.microsoftonline.com/<tenant-id>/v2.0/.well-known/openid-configuration" />
      <required-claims>
        <claim name="aud">
          <value>api://<your-app-registration-id></value>  <!-- * Your app reg client ID -->
        </claim>
      </required-claims>
    </validate-jwt>
    <!-- Extract and forward caller identity -->
    <set-header name="x-caller-objectid" exists-action="override">
      <value>@(context.Request.Headers.GetValueOrDefault("Authorization","").Split(' ').Last().Split('.')[1])</value>
    </set-header>
  </inbound>
</policies>
```

### Request/Response Transformation Policy

```xml
<!-- transformation-policy.xml -->
<policies>
  <inbound>
    <base />
    <!-- Add tracing headers -->
    <set-header name="x-request-id" exists-action="skip">
      <value>@(Guid.NewGuid().ToString())</value>
    </set-header>
    <set-header name="x-forwarded-for" exists-action="override">
      <value>@(context.Request.IpAddress)</value>
    </set-header>
    <!-- Remove sensitive headers before forwarding to backend -->
    <set-header name="Authorization" exists-action="delete" />
    <!-- Add managed identity auth for backend -->
    <authentication-managed-identity resource="https://cognitiveservices.azure.com" />
  </inbound>
  <outbound>
    <base />
    <!-- Remove internal headers from response -->
    <set-header name="x-ms-request-id" exists-action="delete" />
    <set-header name="x-envoy-upstream-service-time" exists-action="delete" />
    <!-- Add CORS headers if not using CORS policy -->
    <set-header name="Access-Control-Allow-Origin" exists-action="override">
      <value>*</value>
    </set-header>
  </outbound>
</policies>
```

## Monitoring Configuration Examples

### Log Analytics Workspace Queries

```kql
// query-ai-usage.kql — Token usage by workload
AzureDiagnostics
| where ResourceType == "SERVICE" and Category == "GatewayLogs"
| where OperationName == "Microsoft.ApiManagement/GatewayLogs"
| extend
    workload = tostring(parse_json(tostring(parse_json(requestBody_s).messages[0])).content),
    product = ProductName,
    tokensConsumed = toint(ResponseHeaders_x_tokens_consumed_s),
    subscriptionId = SubscriptionName
| where isnotempty(tokensConsumed)
| summarize
    TotalTokens = sum(tokensConsumed),
    RequestCount = count(),
    AvgTokensPerRequest = avg(tokensConsumed)
  by product, bin(TimeGenerated, 1h)
| order by TotalTokens desc

// query-error-rates.kql — API error rate by endpoint
ApiManagementGatewayLogs
| where TimeGenerated > ago(1h)
| summarize
    TotalRequests = count(),
    ErrorCount = countif(responseCode >= 400),
    ThrottleCount = countif(responseCode == 429)
  by apiId, bin(TimeGenerated, 5m)
| extend ErrorRate = round(100.0 * ErrorCount / TotalRequests, 2)
| where ErrorRate > 5                     // Alert threshold: >5% errors
| order by ErrorRate desc

// query-content-safety.kql — Content safety block events
AzureDiagnostics
| where Category == "GatewayLogs"
| where statusCode_d == 400
| extend policyError = tostring(parse_json(lastError_source_s))
| where policyError contains "content-safety"
| summarize
    BlockedRequests = count(),
    UniqueSubscriptions = dcount(subscriptionId_s)
  by bin(TimeGenerated, 1h), policyError
```

### Alert Rule Definitions

```bicep
// alert-rules.bicep — Key operational alerts
resource alertTokenQuota 'Microsoft.Insights/scheduledQueryRules@2023-03-15-preview' = {
  name: 'alert-token-quota-high'
  location: location
  properties: {
    displayName: 'AI Token Quota > 90% (1 hour)'
    description: 'Token consumption exceeds 90% of hourly quota'
    enabled: true
    evaluationFrequency: 'PT5M'            // Check every 5 minutes
    windowSize: 'PT1H'                     // Over 1-hour window
    scopes: [logAnalyticsWorkspace.id]
    criteria: {
      allOf: [
        {
          query: '''
            ApiManagementGatewayLogs
            | where TimeGenerated > ago(1h)
            | summarize TokensConsumed = sum(toint(ResponseHeaders_x_tokens_consumed_s))
            | extend QuotaPercent = (TokensConsumed / 100000.0) * 100
            | where QuotaPercent > 90
          '''
          timeAggregation: 'Count'
          operator: 'GreaterThan'
          threshold: 0
          failingPeriods: {
            numberOfEvaluationPeriods: 1
            minFailingPeriodsToAlert: 1
          }
        }
      ]
    }
    actions: {
      actionGroups: [actionGroup.id]
      customProperties: {
        severity: 'warning'
        component: 'ai-gateway'
      }
    }
    severity: 2                            // 0=Critical, 1=Error, 2=Warning
  }
}

resource alertHighErrorRate 'Microsoft.Insights/scheduledQueryRules@2023-03-15-preview' = {
  name: 'alert-apim-error-rate-high'
  location: location
  properties: {
    displayName: 'APIM Error Rate > 5%'
    enabled: true
    evaluationFrequency: 'PT5M'
    windowSize: 'PT15M'
    scopes: [logAnalyticsWorkspace.id]
    criteria: {
      allOf: [
        {
          query: '''
            ApiManagementGatewayLogs
            | summarize ErrorRate = 100.0 * countif(responseCode >= 500) / count()

          operator: 'GreaterThan'
          threshold: 0
        }

    }

  }
```

### Environment-Specific Overrides Pattern

```bicep
// main.bicep — Pattern for environment-driven configuration
@allowed(['dev', 'staging', 'prod'])
param environmentName string

var environmentConfig = {
  dev: {
    apimSku: 'Developer'
    apimCapacity: 1
    enablePrivateEndpoints: false
    logRetentionDays: 30
    enablePurgeProtection: false          // Allow cleanup in dev
    replicaCount: 0
  }
  staging: {
    apimSku: 'Standard'
    apimCapacity: 1
    enablePrivateEndpoints: true
    logRetentionDays: 60
    enablePurgeProtection: true
    replicaCount: 0
  }
  prod: {
    apimSku: 'Premium'
    apimCapacity: 2
    enablePrivateEndpoints: true
    logRetentionDays: 90
    enablePurgeProtection: true
    replicaCount: 1                       // Secondary region replica
  }
}

var config = environmentConfig[environmentName]

// All resources reference var config for environment-specific settings
resource apimService 'Microsoft.ApiManagement/service@2023-05-01-preview' = {
  sku: {
    name: config.apimSku
    capacity: config.apimCapacity
  }
}
```
